# Recommendations from a 9.15 version of gs
# gs -o cmyk-doc.pdf      \
#   -sDEVICE=pdfwrite    \
#   -dOverrideICC=true   \
#   -sDefaultCMYKProfile=/path/to/mycmykprofile.icc \
#   -sOutputICCProfile=/path/to/mydeviceprofile.icc \
#   -dRenderIntent=3     \
#   -dDeviceGrayToK=true \
#    input-doc.pdf
gs -dSAFER -dBATCH -dNOPAUSE -dNOCACHE -sDEVICE=pdfwrite -dAutoRotatePages=/None -sColorConversionStrategy=CMYK -dProcessColorModel=/DeviceCMYK -sOutputFile=document_cmyk.pdf document_rgb.pdf
